from kiteconnect import KiteConnect

def fetch_ohlcv_nse(ticker: str, from_date: str, to_date: str) -> list[dict]:
    kite = KiteConnect(api_key=API_KEY)
    # returns [{date, open, high, low, close, volume}]
    data = kite.historical_data(
        instrument_token=get_token(ticker),
        from_date=from_date,
        to_date=to_date,
        interval="day"
    )
    return normalize_to_kanida_schema(data, market="NSE", ticker=ticker)
